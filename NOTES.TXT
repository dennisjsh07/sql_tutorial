1) CREATING A DATABASE...
CREATE DATABASE test;

2) REMOVE A DATABASE...
DROP DATABASE test;

3) TO USE THE DATABASE CREATED...
USE record_company;

4) CREATE A TABLE...
CREATE TABLE test(
    test_column INT
);

5) ALTER OR ADD COLS TO THE EXISITING TABLE...
ALTER TABLE test
ADD another_col VARCHAR(255);

6) REMOVING THE TABLE...
DROP TABLE test;

7) REMOVING THE COL...
ALTER TABLE test
DROP another_col;
